// server/services/dmBufferService.ts
interface PendingBuffer {
  messages: Message[];
  timer: NodeJS.Timeout;
  conversationId: string;
  brandId: string;
}
class DmBufferService {
  private buffers: Map<string, PendingBuffer> = new Map();
  private BUFFER_DELAY_MS = 15000; // 15 segundos
  // Llamar cuando llega un DM
  async bufferMessage(message: Message, conversation: Conversation, brand: Brand): Promise<void> {
    // Solo aplicar a DMs
    if (message.type !== 'conversation') {
      // Comentarios: procesar inmediatamente
      await autoReplyService.processNewMessage(message, conversation, brand);
      return;
    }
    const key = `${conversation.id}`; // Buffer por conversación
    const existing = this.buffers.get(key);
    if (existing) {
      // Cancelar timer anterior
      clearTimeout(existing.timer);
      // Acumular mensaje
      existing.messages.push(message);
      // Reiniciar timer
      existing.timer = setTimeout(() => this.flushBuffer(key), this.BUFFER_DELAY_MS);
    } else {
      // Crear nuevo buffer
      const timer = setTimeout(() => this.flushBuffer(key), this.BUFFER_DELAY_MS);
      this.buffers.set(key, {
        messages: [message],
        timer,
        conversationId: conversation.id,
        brandId: brand.id,
      });
    }
  }
  private async flushBuffer(key: string): Promise<void> {
    const buffer = this.buffers.get(key);
    if (!buffer) return;
    this.buffers.delete(key);
    // Obtener datos frescos
    const conversation = await storage.getConversation(buffer.conversationId);
    const brand = await storage.getBrand(buffer.brandId);
    
    // Usar el ÚLTIMO mensaje como el "trigger", pero pasar TODOS como contexto
    const lastMessage = buffer.messages[buffer.messages.length - 1];
    
    // El historial ya incluirá todos los mensajes acumulados
    await autoReplyService.processNewMessage(lastMessage, conversation, brand);
  }
}

// En prompt-composer.ts
interface SituationContext {
  messageType: 'dm' | 'comment';
  relationshipStatus: 'new' | 'active' | 'reengagement';
  conversationDepth: number;
  timeSinceLastInteraction: number; // minutos
  messagesLastHour: number;
  situationInstructions: string;
}
function calculateSituationContext(
  message: Message,
  conversationHistory: Message[]
): SituationContext {
  const messageType = message.type === 'conversation' ? 'dm' : 'comment';
  
  // Profundidad de la conversación
  const conversationDepth = conversationHistory?.length || 0;
  
  // Mensajes en la última hora
  const oneHourAgo = Date.now() - (60 * 60 * 1000);
  const messagesLastHour = conversationHistory?.filter(
    m => new Date(m.timestamp).getTime() > oneHourAgo
  ).length || 0;
  
  // Tiempo desde última respuesta nuestra
  const lastOutbound = conversationHistory
    ?.filter(m => m.direction === 'outbound')
    .sort((a, b) => new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime())[0];
  
  const timeSinceLastInteraction = lastOutbound 
    ? Math.floor((Date.now() - new Date(lastOutbound.timestamp).getTime()) / 60000)
    : 999;
  
  // Determinar estado de la relación
  let relationshipStatus: 'new' | 'active' | 'reengagement';
  if (conversationDepth === 0 || conversationDepth === 1) {
    relationshipStatus = 'new';
  } else if (timeSinceLastInteraction < 60) {
    relationshipStatus = 'active';
  } else {
    relationshipStatus = 'reengagement';
  }
  
  // Generar instrucciones automáticas basadas en la situación
  let situationInstructions = '';
  
  if (messageType === 'dm') {
    if (relationshipStatus === 'new') {
      situationInstructions = 'Primera interacción. Saluda brevemente y preséntate.';
    } else if (relationshipStatus === 'active') {
      situationInstructions = 'Conversación activa. PROHIBIDO saludar. Ve directo al grano.';
    } else {
      situationInstructions = 'Retomando conversación. Puedes saludar suavemente ("Hola de nuevo").';
    }
    
    if (messagesLastHour > 5) {
      situationInstructions += ' Usuario muy activo, respuestas ultra cortas.';
    }
  }
  
  return {
    messageType,
    relationshipStatus,
    conversationDepth,
    timeSinceLastInteraction,
    messagesLastHour,
    situationInstructions,
  };
}

  isValid: boolean;
  issue?: string;
} {

  if (situation.relationshipStatus === 'active') {

      };
    }
  }
  
  return { isValid: true };
}
